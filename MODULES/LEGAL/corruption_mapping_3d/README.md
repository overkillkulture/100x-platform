```python
    shell_indicators = []

    for company in network.companies:
        score = 0

        # No employees
        if company.employees == 0:
            score += 0.3

        # Same address as other companies
        same_address_count = count_companies_at_address(company.address)
        if same_address_count > 5:
            score += 0.2

        # Nominee director (serves on many boards)
        if company.director_board_count > 10:
            score += 0.2

        # Large revenue, no operations
        if company.revenue > 1000000 and company.employees == 0:
            score += 0.3

        if score > 0.6:
            shell_indicators.append({
                'company': company,
                'confidence': score,
                'reason': 'Multiple shell company indicators'
            })

    return shell_indicators
```

### Suspicious Timing Detection:
```python
def detect_suspicious_timing(network):
    """Find payments with suspicious timing"""

    suspicious = []

    for payment in network.payments:
        # Payment right before contract award
        for contract in network.contracts:
            days_before = (contract.date - payment.date).days

            if 0 < days_before < 30:
                if payment.from_entity in contract.related_entities:
                    suspicious.append({
                        'payment': payment,
                        'contract': contract,
                        'days_before': days_before,
                        'suspicion_level': 'HIGH',
                        'reason': f'Payment {days_before} days before contract'
                    })

    return suspicious
```
